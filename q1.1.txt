public boolean isUnique(String string){
    /*
        naive:
            iterate through all characters in the string and compare them wiht every otehr charatcer 
                --> naive o(n^2) sol'n
    */
    for (int i = 0; i < string.length-1; i++){
        char x = string.charAt(i);

        for (int j = i+1; j< string.length; j++){
            if (x.equals(string.charAt(j))) return false;
        }
    }
    return true;
}
